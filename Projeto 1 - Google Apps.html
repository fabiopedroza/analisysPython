# Função para realizar as conversões de tipo
def convert_size(size):
    if size is not np.nan:
        if 'k' in size:
            return float(size.replace('k','')) * 1000
        if 'M' in size:
            return float(size.replace('M','')) * 1000000
        
# Aplica ao tamanho
df['Size'] = df['Size'].apply(convert_size)

    if '$' in str(price):
